\section{Introdução}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Introdução}
	\vspace{10px}

	\textbf{Redes de Sensores sem Fio (RSSF)}
	\begin{itemize}
		\justifying
		\item{Conjunto de nodos espalhados aleatoriamente com capacidade de auto-gerenciamento \cite{akyildiz2:2002};}
		\item{Objetivo: monitorar algum fenômeno no ambiente (esforço colaborativo);}
		\item{Aplicações: indústria, meio ambiente, militar, saúde, etc.;}
		\item{Problema: gerenciamento energético, escalonamento do sono, etc.;}
	\end{itemize}
\end{frame}



\section{Fundamentação Teórica}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Fundamentação Teórica}
	\vspace{10px}

	\textbf{Baterias}
	\begin{itemize}
		\justifying
		\item{Dispositivo capaz de transformar energia química em elétrica;}
		\item{Componentes: Ânodo, Cátodo, Eletrólito e eletrodos;}
		
		\begin{center}
			\includegraphics[width=0.45\textwidth]{battery.pdf}
		\end{center}
		
		 \item{Tecnologias: Ni-Cd, Ni-MH, Li-ion, Li-polymer, etc.;}
	\end{itemize}
\end{frame}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Fundamentação Teórica}
	\vspace{10px}

	\textbf{Baterias: Efeitos intrínsecos}
	\begin{itemize}
		\justifying
		\item{Efeito da Taxa de Capacidade (\emph{Rate Capacity Effect}):}
		\begin{itemize}
			\justifying
			\item{Dependência entre capacidade atual e a magnitude da corrente aplicada;}
			\item{Ex.: altas correntes de descarga reduzem significativamente a capacidade;}
		\end{itemize}
		
		\item{Efeito Recuperação (\emph{Recovery Effect}):}
		\begin{itemize}
			\justifying
			\item{Capacidade de recuperar carga quando nenhuma carga externa está conectada;}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Fundamentação Teórica}
	\vspace{10px}

	\textbf{Modelos de Baterias}
	\begin{itemize}
		\justifying
		\item{Objetivos:}
		\begin{itemize}
			 \item{Modelar o comportamento (não linear) das baterias;}
			 \item{Estimar o tempo de vida de uma bateria de acordo com as cargas aplicadas;}
		 \end{itemize}
		 \item{Tipos: Eletroquímico, Circuito Elétrico, Analíticos, Estocásticos, Híbridos;}
	\end{itemize}
	\vspace{10px}
	\textbf{Modelo Analítico: Kinect Battery Model (KiBaM) \cite{manwell:1993}}
	\begin{itemize}
		\justifying
		\item{Modelo abstrato e bastante intuitivo;}
		\item{Dois tanques: Carga Disponível e Carga Limite;}
	\end{itemize}
\end{frame}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Fundamentação Teórica}
	\vspace{10px}
	
	\textbf{Modelo Analítico: Kinect Battery Model (KiBaM) \cite{manwell:1993}}
	\begin{itemize}
		\justifying
		\item{Fluxo de carga depende dos parâmetros: ($h_2 - h_1$) e $k$;}
		\item{Modela adequadamente os Efeitos Taxa de Capacidade e Recuperação \cite{jongerden:2008-2};}
	\end{itemize}
	\begin{center}
		\includegraphics[width=0.55\textwidth]{kibam-model.pdf}
	\end{center}
\end{frame}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Fundamentação Teórica}
	\vspace{10px}

	\textbf{Modelo Analítico: Kinect Battery Model (KiBaM) \cite{manwell:1993}}
	
	\begin{equation}
		\begin{cases}
			i = i_0e^{-k't} + \frac{(y_0k'c-I)(1-e^{-k't})}{k'} - \frac{Ic(k't-1+e^{-k't})}{k'} \\
			j = j_0e^{-k't} + y_0(1-c)(1-e^{-k't}) - \frac{I(1-c)(k't-1+e^{-k't})}{k'},
		\end{cases}
		\label{eq:kibam-equation}
	\end{equation}
	\vspace{10px}
	\begin{equation*}
		k' = \frac{k}{c(1-c)}, \hspace{10px} y0 = i0 + j0;
	\end{equation*}
\end{frame}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Fundamentação Teórica}
	\vspace{10px}

	\textbf{Escalonamento do Sono (\textit{Sleep Scheduling})}
	\begin{itemize}
		\justifying
		\item{Objetivos:}
		\begin{itemize}
			 \item{Realizar o rodízio entre os nodos para poupar energia;}
			 \item{Escolher adequadamente a ordem de execução das tarefas;}
			 \item{Aumentar o tempo de vida da rede;}
		 \end{itemize}
		 \item{Tarefas que precisam de maior carga devem executar primeiro \cite{rakhmatov:2002};}
		 \item{Inserir períodos ociosos entre tarefas favorece a bateria \cite{li:2013};}
	\end{itemize}
\end{frame}

		\item{Objetivos específicos do trabalho:}
		\begin{itemize}
			\justifying
			\item{Observar o aumento no tempo de vida de uma RSSF quando o Efeito Recuperação faz parte da rotina dos nodos;}

\section{Conclusão}

\begin{frame}
	\vspace{-15pt}
	\frametitle{Conclusão}
	\vspace{10px}
	
	\begin{itemize}
		\justifying
		\item{O consumo energético dos nodos é relevante no contexto das RSSF;}
		\item{O Efeito da Taxa de Capacidade e, principalmente, o Efeito Recuperação desempenham importante papel no tempo de vida da bateria de um nodo;}
		\item{Em uma RSSF com diversos nodos, alguns podem coletar informações enquando outros recuperam energia;}
	\end{itemize}
	
	\vspace{10px}
	\textbf{Trabalhos Futuros}
	\begin{itemize}
		\justifying
		\item{Expandir as simulações para diversos nodos.}
		\item{Avaliar o desempenho do KiBaM em baterias Li-ion.}
		\item{Integrar o modelo de bateria em uma ferramenta voltada para simulação de RSSF, como o OMNeT++;}
	\end{itemize}
\end{frame}
